# Steps to Run on Windows

cd app 

python -m venv venv  

venv\Scripts\activate

pip install -r requirements.txt

python main.py


# Steps to Run for macOS

cd app 

python3 -m venv venv

source venv/bin/activate

pip install -r requirements.txt

python main.py


# Should see output from SBERT in terminal and new Tkinter window
